    <script>
        async function envoyer() {
            const input = document.getElementById('fichierPdf');
            const prixInput = document.getElementById('prixInitial');
            const conteneur = document.getElementById('contenu-rapport');
            const resultSection = document.getElementById('resultats');
            
            if (!input.files.length) return alert("Veuillez sélectionner un dossier technique (PDF).");
            
            conteneur.innerHTML = "<p><i>Analyse juridique et financière en cours... Veuillez patienter.</i></p>";
            resultSection.style.display = "block";

            const formData = new FormData();
            formData.append("fichier", input.files[0]);
            formData.append("prix", prixInput.value || 0);

            try {
                const reponse = await fetch("https://audit-check.onrender.com/scan", {
                    method: "POST",
                    body: formData
                });
                
                const data = await reponse.json();
                let html = `<p style="font-size: 11px; color: #27ae60; text-align: right;">${data.securite}</p>`;
                
                if (data.anomalies.length > 0) {
                    html += `
                    <div class="total-box">
                        <div style="font-size:14px; opacity:0.8;">VALEUR NETTE ESTIMÉE APRÈS TRAVAUX</div>
                        <div style="font-size:24px; font-weight:bold;">${data.prix_net.toLocaleString()} €</div>
                        <div style="font-size:13px; color:#e74c3c;">Décote totale : -${data.total_decote} €</div>
                    </div>`;
                    
                    html += `<h3>⚠️ Points de vigilance (Non-conformités)</h3>`;
                    data.anomalies.forEach(a => {
                        html += `<div class="card-danger">
                            <strong>${a.titre}</strong><br>
                            <span>${a.detail}</span><br>
                            <span style="color:#e74c3c; font-weight:bold;">Budget travaux : -${a.cout}€</span>
                            <span class="loi">Référence légale : ${a.loi}</span>
                        </div>`;
                    });
                } else {
                    html += `<div class="total-box" style="background:#27ae60;">
                        <div style="font-size:14px; opacity:0.8;">VALEUR PRÉSERVÉE</div>
                        <div style="font-size:24px; font-weight:bold;">${data.prix_initial.toLocaleString()} €</div>
                        <div style="font-size:13px;">Aucun travaux majeurs requis</div>
                    </div>`;
                }

                if (data.positives.length > 0) {
                    html += `<h3>✅ Atouts valorisables</h3>`;
                    data.positives.forEach(p => {
                        html += `<div class="card-success">✔️ ${p}</div>`;
                    });
                }

                conteneur.innerHTML = html;

            } catch (e) {
                conteneur.innerHTML = "<span style='color:#e74c3c; font-weight:bold;'>Erreur de connexion au serveur sécurisé.</span>";
            }
        }
    </script>
